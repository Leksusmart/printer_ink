Страница создания заявки:

Страница автоматически считывает номер телефона сотрудника из адресной строки, делает GET-запрос к бэкенду и выводит реальное ФИО текущего пользователя в верхнюю неизменяемую графу.
{
  "type": "Приёмка",          // Либо "Выдача" в зависимости от выбранного тумблера операций
  "isDeflective": false,      // Булево значение: true (если есть брак / выбран тумблер "Нет"), false (если исправен / "Да")
  "status": "Создана",        // Стандартный начальный статус для новой заявки
  "data": "2026-07-10 17:22:47", // Текущие дата и время создания с точностью до секунды
  "employee": 42,             // ID реального сотрудника из таблицы employers, полученный при авторизации
  "lastchangedata": "2026-07-10 17:22:47", // Время последнего изменения (совпадает с датой создания)
  "lastchangeby": 42,         // ID сотрудника, изменившего заявку (совпадает с создателем)
  "comment": "Текст комментария пользователя", // Текст примечания (или пустая строка)
  "model": "HP LJ 1010",
  "amount": 2
  
}
